# 5. Циклы, логические конструкции

### 5.1. Условия

```
if (a === 1) {
    console.log(1);
}
else if (a === 2) {
    console.log(2);
}
else {
    console.log(3);
}
```

```
switch(a) {
    case 1:
        console.log(1);
        break;
    case 2:
        console.log(2);
        break;
    case 3:
        console.log(3);
    case 4:
        console.log(4);
        break;
    default:
        console.log('default');
        break;
}
```

### 5.2. Циклы

```
while (--a) {
    console.log(a);
}
```

```
do {
    console.log(a);
}
while (--a)
```

```
for (let i = 0; i < 10; i++) {
    console.log(a);
}
```

```
for (let item of [1, 2, 3]) {
    console.log(item);
}
```

```
for (let field in {a: 1, b: 2, c: 3}) {
    console.log(item);
}
```
